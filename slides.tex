\section{Feature Engineering}

\begin{frame}{Feature Engineering}
  \begin{columns}[T]
    \begin{column}{0.32\textwidth}
      \begin{block}{Lag Features}
        Giá trị quá khứ của từng chất ô nhiễm tại các mốc:
        \vspace{0.3em}
        \begin{center}
          \textbf{--1h · --3h · --6h}\\
          \textbf{--12h · --24h · --168h}
        \end{center}
        \vspace{0.3em}
        \small 7 biến $\times$ 6 lags = 42\\
        Weather: 16 $\times$ 3 = 48
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{Rolling Features}
        Mean \& Std trượt của AQI, PM2.5, PM10:
        \vspace{0.3em}
        \begin{center}
          \textbf{3h · 6h · 24h · 168h}
        \end{center}
        \vspace{0.3em}
        \small 3 biến $\times$ 4 windows $\times$ 2 (mean/std) = 24
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{Time Features}
        Mã hóa chu kỳ:
        \begin{itemize}\small
          \item \texttt{hour\_sin/cos}
          \item \texttt{weekday\_sin/cos}
          \item \texttt{month\_sin/cos}
          \item \texttt{is\_weekend}
          \item \texttt{is\_dry\_season}
        \end{itemize}
        \small Mùa khô: tháng 11--4
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \begin{alertblock}{Tại sao cần lag 168h (1 tuần)?}
    Mô hình học được pattern lặp lại theo tuần --- giờ cao điểm thứ Hai sẽ tương tự thứ Hai tuần trước.
    Cần \textbf{ít nhất 170h lịch sử} để tính đủ rolling 168h.
  \end{alertblock}
\end{frame}

\section{Các mô hình}

\begin{frame}{Các mô hình sử dụng}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{MARS (Multivariate Adaptive Regression Splines)}
        \begin{itemize}\small
          \item Tự động tìm \textbf{điểm gãy (knots)} trong dữ liệu
          \item Phù hợp với quan hệ phi tuyến cục bộ
          \item Thư viện: \texttt{py-earth2}
          \item Pipeline: StandardScaler $\to$ Earth
        \end{itemize}
      \end{block}
      \vspace{0.3em}
      \begin{block}{SVMR (Support Vector Regression)}
        \begin{itemize}\small
          \item Kernel RBF, epsilon-insensitive loss
          \item Grid search: C, epsilon, gamma
          \item Pipeline: StandardScaler $\to$ SVR
          \item Regularization mạnh, ổn định ở horizon dài
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{MLP (Neural Network Regressor)}
        \begin{itemize}\small
          \item Mạng nơ-ron nhiều lớp ẩn
          \item Activation: ReLU, Optimizer: Adam
          \item Pipeline: StandardScaler $\to$ MLPRegressor
          \item Max iterations: 500
        \end{itemize}
      \end{block}
      \vspace{0.3em}
      \begin{block}{ARIMA}
        \begin{itemize}\small
          \item Mô hình chuỗi thời gian cổ điển
          \item Univariate (chỉ dùng US AQI)
          \item Grid search order (p $\leq$ 3, d $\leq$ 1, q $\leq$ 3)
          \item Chọn theo AIC $\to$ ARIMA(3,0,3)
        \end{itemize}
      \end{block}
      \vspace{0.3em}
      \textbf{Baseline:} Persistence --- dự đoán AQI tương lai = AQI hiện tại.
    \end{column}
  \end{columns}
\end{frame}
